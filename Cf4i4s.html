문제

사용자가 차를 주문하면 따뜻한 차를 출력하는 문제입니다 실행순서를 클릭하세요 
[참고] 녹차 \033[42m 레몬차 \033[43m 홍차 \033[41m 색상 초기화 \033[0m
물위치 \033[3;3f \033[4;3f \033[5;3f
녹차(1), 레몬차(2), 홍차(3) 중 하나를 숫자로 입력하세요: 2
 |         |\
 |         |/
 |____|

프로그램 코드

	#include <stdio.h>

		

1	int main() {

2		int tea = 0;


3		printf( "녹차(1), 레몬차(2), 홍차(3) 중 하나를 숫자로 입력하세요: ");

4		scanf( "%d", &tea );


5		printf("\n |    |\\");

6		printf("\n |    |/");

7		printf("\n |____|");

		

8		switch ( tea ) {

9			case 1:

10				printf(" \033[42m");

11				break;

12			case 2:

13				printf(" \033[43m");

14				break;

15			default:

16				printf(" \033[41m");

		}

		

17		printf(" \033[3;3f    ");

18		printf(" \033[4;3f    ");

19		printf(" \033[5;3f____ \033[0m");


20		return 0;

	}
